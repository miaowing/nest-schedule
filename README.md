## Description

This is a [Nest](https://github.com/nestjs/nest) module for using decorator schedule a job.

## Installation

```bash
$ npm i --save nest-schedule
```

## Quick Start

```typescript
import { Injectable } from '@nestjs/common';
import { Schedule } from 'nest-schedule';

@Injectable()
export class ScheduleService {  
  @Schedule({
    cron: '0 0 2 * *',
    startTime: new Date(), 
    endTime: new Date(new Date().getTime() + 24 * 60 * 60 * 1000)
  })
  syncData() {
    console.log('syncing data ...');
  }
  
  @Schedule({cron: '0 0 4 * *'})
  clear() {
    console.log('syncing data ...');
  }
  
  @Schedule({timeout: 5000})
  onceJob() {
    console.log('once job');
  }
  
  @Schedule({interval: 2000})
  intervalJob() {
    console.log('interval job');
    
    // if you want to cancel the schedule job, you should return true;
    return true;
  }
}
```
